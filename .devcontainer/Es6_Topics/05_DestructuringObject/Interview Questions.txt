1. Renaming properties
    const user = { name: "Mohit", age: 28 };
    const { name: userName, age: userAge } = user;
    console.log(userName, userAge);

2. Default values
    const settings = { theme: "dark" };
    const { theme, language = "English" } = settings;
    console.log(theme, language);

3. Nested objects
    const response = { data: { product: { title: "Laptop", price: 45000 } } };
    const { data: { product: { title, price } } } = response;
    console.log(title, price);

4. Function parameters
    function showUser({ name, age }) {
      console.log(`Name: ${name}, Age: ${age}`);
    }
    showUser({ name: "Riya", age: 25 });

5. React props best practice
    function Profile({ name, age }) {
  return <h2>{name} - {age}</h2>;
}
